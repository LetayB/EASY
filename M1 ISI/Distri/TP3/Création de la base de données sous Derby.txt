-- === drop statements ===
DROP TABLE DBUSER.SEQUENCE;
DROP TABLE DBUSER.ARTICLE_COMMENTAIRE;
DROP TABLE DBUSER.ARTICLE;
DROP TABLE DBUSER.COMMENTAIRE;
DROP SCHEMA DBUSER RESTRICT;
-- === create statements ===
CREATE SCHEMA DBUSER;
CREATE TABLE DBUSER.SEQUENCE (
	SEQ_NAME VARCHAR (50) NOT NULL,
	SEQ_COUNT DECIMAL (5),
	CONSTRAINT SEQ_NAME_PK PRIMARY KEY (SEQ_NAME)
);
CREATE TABLE DBUSER.ARTICLE (
	ID BIGINT NOT NULL,
	AUTEUR VARCHAR (255) NOT NULL,
	CDATE TIMESTAMP,
	CONTENU VARCHAR (255) NOT NULL,
	TITRE VARCHAR (255) NOT NULL,
	CONSTRAINT ARTICLE_ID_PK PRIMARY KEY (ID)
);
CREATE TABLE DBUSER.COMMENTAIRE (
	ID BIGINT NOT NULL,
	CDATE TIMESTAMP,
	CONTENU VARCHAR (255) NOT NULL,
	EMAIL VARCHAR (255) NOT NULL,
	CONSTRAINT COMMENTAIRE_ID_PK PRIMARY KEY (ID)
);
CREATE TABLE DBUSER.ARTICLE_COMMENTAIRE (
	ARTICLE_ID BIGINT NOT NULL,
	COMMENTAIRES_ID BIGINT NOT NULL,
	CONSTRAINT ARTICLE_COMMENTAIRE_ARTICLE_ID_COMMENTAIRES_ID PRIMARY KEY (ARTICLE_ID, COMMENTAIRES_ID)
);
-- === alter statements ===
ALTER TABLE DBUSER.ARTICLE_COMMENTAIRE ADD CONSTRAINT ARTICLE_COMMENTAIRE_ARTICLE_ID_FK FOREIGN KEY (ARTICLE_ID) REFERENCES DBUSER.ARTICLE ON DELETE NO ACTION ON UPDATE NO ACTION;
ALTER TABLE DBUSER.ARTICLE_COMMENTAIRE ADD CONSTRAINT ARTICLE_COMMENTAIRE_COMMENTAIRES_ID_FK FOREIGN KEY (COMMENTAIRES_ID) REFERENCES DBUSER.COMMENTAIRE ON DELETE NO ACTION ON UPDATE NO ACTION;
